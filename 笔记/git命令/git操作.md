git init - 初始化仓库。
git add . - 添加文件到暂存区。
git commit - 将暂存区内容添加到仓库中。

git branch - 查看分支
git branch (branchname) - 创建分支
git checkout (branchname) - 切换分支
git merge - 合并分支

git clone (仓库地址) - 克隆仓库
git log - 查看日志
git reset (filename) - 回退
git reset (commitID) - 版本回退
git alias - 自定义命令

# 提交过程
git add .
git commit -m (message)
git push

# 拉取代码
git checkout (分支名)
git pull
